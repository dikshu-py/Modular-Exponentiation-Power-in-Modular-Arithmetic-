try:
    import sys


    # Iterative Python3 program
    # to compute modular power

    # Iterative Function to calculate
    # (x^y)%p in O(log y)
    def power(x, y, p):
        res = 1  # Initialize result

        # Update x if it is more
        # than or equal to p
        x = x % p

        if (x == 0):
            return 0

        while (y > 0):

            # If y is odd, multiply
            # x with result
            if ((y & 1) == 1):
                res = (res * x) % p

            # y must be even now
            y = y >> 1  # y = y/2
            x = (x * x) % p

        return res


    test_cases = sys.stdin.readline()
    for i in range(int(test_cases)):

        print(power(2,int(input())-1,1000000007))
